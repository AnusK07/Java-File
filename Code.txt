<div class="box">
  <h3>Mouse Event Demonstrations</h3>
  <div class="event hover">Hover over me to change color</div>
  <div class="event dblclick">Double-click me to hide</div>
  <div class="event press">Press mouse down here to grow</div>
  <div class="event style">This box will change style</div>
  <button id="change">Change Style</button>
</div>
<script>
  $(function(){
    $(".hover").hover(function(){ $(this).css("background","#d1e7dd") }, function(){ $(this).css("background","") });
    $(".dblclick").dblclick(function(){ $(this).hide() });
    $(".press").on("mousedown", function(){ $(this).css("transform","scale(1.2)") })
               .on("mouseup", function(){ $(this).css("transform","") });
    $("#change").click(function(){
      $(".style").css({ color:"#fff", background:"#333", border:"2px solid #000" });
    });
  });
</script>
